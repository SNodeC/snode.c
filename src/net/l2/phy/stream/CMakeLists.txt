cmake_minimum_required(VERSION 3.18)

set(NET-L2-PHY-STREAM_CPP PhysicalSocket.cpp PhysicalSocketClient.cpp
                          PhysicalSocketServer.cpp
)

set(NET-L2-PHY-STREAM_H PhysicalSocketClient.h PhysicalSocketServer.h
                        PhysicalSocket.hpp PhysicalSocket.h
)

add_library(
    net-l2-phy-stream SHARED ${NET-L2-PHY-STREAM_CPP} ${NET-L2-PHY-STREAM_H}
)
add_library(snodec::net-l2-phy-stream ALIAS net-l2-phy-stream)

target_link_libraries(net-l2-phy-stream PUBLIC net-l2-phy)

target_include_directories(
    net-l2-phy-stream PUBLIC "$<BUILD_INTERFACE:${PROJECT_SOURCE_DIR}>"
                             "$<INSTALL_INTERFACE:include/snode.c>"
)

set_target_properties(
    net-l2-phy-stream PROPERTIES SOVERSION ${SNODEC_SOVERSION}
                                 OUTPUT_NAME snodec-net-l2-phy-stream
)

install(
    TARGETS net-l2-phy-stream
    EXPORT snodec_net-l2-phy-stream_Targets
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR} COMPONENT net-l2-stream
)

install(
    DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/
    DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}/snode.c/net/l2/phy/stream
    COMPONENT net-l2-phy-stream
    FILES_MATCHING
    PATTERN "*.h"
    PATTERN "*.hpp"
    PATTERN "cmake" EXCLUDE
    PATTERN "legacy" EXCLUDE
    PATTERN "tls" EXCLUDE
)

install(
    EXPORT snodec_net-l2-phy-stream_Targets
    FILE snodec_net-l2-phy-stream_Targets.cmake
    NAMESPACE snodec::
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/snodec
    COMPONENT net-l2-phy-stream
)
